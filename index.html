              <script>
                new Vue({
                  el: '#app',
                  data: {
                    message: 'Hello MiamiJS',
                  }
                  computed: {
                    messageOnSteroids: function() {
                      var chars = this.message.split('')
                      var steroidsMessage = ''

                      for (var i=0; i < chars.length; ++i) {
                        if (i % 2 == 0) {
                          steroidsMessage += chars[i].toUpperCase()
                        } else {
                          steroidsMessage += chars[i].toLowerCase()
                        }
                      }

                      return steroidsMessage
                    }
                  }
                })
              </script>
